New submissions

 

Please send them to Daniel Schneider, Daniel.Schneider@tecfa.unige.ch. Submit in BibTex Format if you can! Put the URL in the note and don't worry about small errors: something along the ``BibTex line'' is still preferrable. Here are the most frequent Bibtex styles: 

@Article{,
  author =       "",
  title =        "",
  OPTcrossref =  "",
  OPTkey =       "",
  OPTjournal =   "",
  OPTyear =      "",
  OPTvolume =    "",
  OPTnumber =    "",
  OPTpages =     "",
  OPTmonth =     "",
  OPTnote =      "",
  OPTannote =    ""
}

@InProceedings{,
  author =       "",
  title =        "",
  OPTcrossref =  "",
  OPTkey =       "",
  OPTeditor =    "",
  OPTvolume =    "",
  OPTnumber =    "",
  OPTseries =    "",
  OPTpages =     "",
  OPTbooktitle = "",
  OPTyear =      "",
  OPTorganization = "",
  OPTpublisher = "",
  OPTaddress =   "",
  OPTmonth =     "",
  OPTnote =      "",
  OPTannote =    ""
}

@TechReport{,
  author =       "",
  title =        "",
  institution =  "",
  year =         "",
  OPTcrossref =  "",
  OPTkey =       "",
  OPTtype =      "",
  OPTnumber =    "",
  OPTaddress =   "",
  OPTmonth =     "",
  OPTnote =      "",
  OPTannote =    ""
}

@Misc{,
  OPTcrossref =  "",
  OPTkey =       "",
  OPTauthor =    "",
  OPTtitle =     "",
  OPThowpublished = "",
  OPTyear =      "",
  OPTmonth =     "",
  OPTnote =      "",
  OPTannote =    ""
}
